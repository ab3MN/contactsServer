    User model:

  • password: String required
  • email: String required
  • subscription: String
    enum: ['starter', 'pro', 'business']
    default: 'starter'
  • largerAvatarURL: String
    default:img size 250px
  • smallAvatarURL: String
    default:img size 80px
  • isActivated: Boolean
    default: false
  • activationLink: String
    default: ''
  • role: String
    default: 'user'

    await email password

  • if no email or password return status 409
  • if email in use return status 409

  • if wrong email or password return status 401
